git clone <link>
git status //displays status of code
git add <file name> or git add . //to add all untracked files
git commit -m "message"
git push origin main //upload from local repo to remote repo


git init //create new git repo
git remote add origin <-link->
git remote -v (to verify remote)
git branch (to check branch)
git branch -M main (to rename branch from master to main)
git push origin main

//BRANCH COMMands
git checkout <branch name> //to navigate
git checkout -b <new branch name>  //create new branch
git checkout main //to come back to main branch
git branch -d <branch name> //delete branch
